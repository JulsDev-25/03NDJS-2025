Syntaxe bouble for
for (let depart; fin; incrément) {
    ..instructions
}

//Exemple de bouble for
for (let i=0; i<11; i++) {
    console.log(i);
}


Syntaxes de boucle do.. wille
let depart = valeur;
do {
    incrément;
    ..instructions;
    
} while (fin)

//exemple de boucle do.. wille
let i = 0;
do {
    i += 1;
    console.log(i);
    
} while (i<5)

Syntaxes de boucle while
let depart = valeur;

while (fin_boucle) {
  ..instructions;
  incrément;
}
